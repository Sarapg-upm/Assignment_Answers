def fetch(url)
  response = RestClient::Request.execute({
    method: :get,
    url: url.to_s})
  if response.body
    return response.body
  else
    return false
  end
end

def write_csv(path, data, sep = "\t", header)
  csv = CSV.open(path, "w", :write_headers=> true, :headers => header, :col_sep => sep)
  for row in data 
    csv << row
  end
end
